sortObjectsByDist <- function(xl, z, metricFunction = euclideanDistance)#–∑–∞–¥–∞–µ–º —Ñ—É–Ω–∫—Ü–∏—é —Ä–∞—Å—Å—Ç–æ—è–Ω–∏—è
{
  l <- dim(xl)[1] 
  n <- dim(xl)[2] - 1
  distances <- matrix(NA, l, 2) #–∑–∞–¥–∞–µ–º –º–∞—Ç—Ä–∏—Ü—É —Ä–∞—Å—Å—Ç–æ—è–Ω–∏—è
  for (i in 1:l) 
  {
    distances[i,] <- c(i, metricFunction(xl[i, 1:n], z)) ## —Å—á–∏—Ç–∞–µ–º —Ä–∞—Å—Å—Ç–æ—è–Ω–∏–µ –æ—Ç –∫–ª–∞—Å—Å–∏—Ñ–∏—Ü–∏—Ä—É–µ–º–æ–π —Ç–æ—á–∫–∏ –¥–æ –æ—Å—Ç–∞–ª—å–Ω—ã—Ö —Ç–æ—á–µ–∫ –≤—ã–±–æ—Ä–∫–∏
  }
  orderedXl <- xl[order(distances[, 2]),] ##—Å–æ—Ä—Ç–∏—Ä—É–µ–º —Å–æ–≥–ª–∞—Å–Ω–æ –ø–æ—Å—á–∏—Ç–∞–Ω–æ–≥–æ —Ä–∞—Å—Å—Ç–æ—è–Ω–∏—è
  return (orderedXl)
}
```

{

  return (class) ## –≤–æ–∑–≤—Ä–∞—â–∞–µ–º –∫–ª–∞—Å—Å

```

LOOPW = function(points, classes, hValues) {
  n = dim(points)[1]
  loo = rep(0, length(hValues))
  
  for (i in 1:n) {
    u = points[i,]
    sample = points[-i,]
    distances = distances(sample, u)
    names(distances) = classes[-i]
    
    for (j in 1:length(hValues)) {
      h = hValues[j]
      classified = PW(distances, u, h)
      loo[j] = loo[j] + (classified != classes[i])
    }
  }
  
  loo = loo / n
}

potentialFunction <- function(distances, potentials, h, xl, type_core) {
  l <- nrow(xl)
  n <- ncol(xl)
  classes <- xl[, n]
  weights <- table(classes) # –¢–∞–±–ª–∏—Ü–∞ –¥–ª—è –≤–µ—Å–æ–≤ –∫–ª–∞—Å—Å–æ–≤
  weights[1:length(weights)] <- 0 # –ü–æ —É–º–æ–ª—á–∞–Ω–∏—é –≤—Å–µ –≤–µ—Å–∞ —Ä–∞–≤–Ω—ã –Ω—É–ª—é
  for (i in 1:l) { # –î–ª—è –∫–∞–∂–¥–æ–≥–æ –æ–±—ä–µ–∫—Ç–∞ –≤—ã–±–æ—Ä–∫–∏
    class <- xl[i, n] # –ë–µ—Ä–µ—Ç—Å—è –µ–≥–æ –∫–ª–∞—Å—Å
    r <- distances[i] / h[i]
    weights[class] <- weights[class] + potentials[i] * type_core(r) # –°—á–∏—Ç–∞–µ—Ç—Å—è –µ–≥–æ –≤–µ—Å, –∏ –ø—Ä–∏–±–∞–≤–ª—è–µ—Ç—Å—è –∫ –æ–±—â–µ–º—É –≤–≤–µ—Å—É –µ–≥–æ –∫–ª–∞—Å—Å–∞
  }
  if (max(weights) != 0) return (names(which.max(weights))) # –ï—Å–ª–∏ –µ—Å—Ç—å –≤–µ—Å –±–æ–ª—å—à–µ –Ω—É–ª—è, —Ç–æ –≤–µ—Ä–Ω—É—Ç—å –∫–ª–∞—Å—Å —Å –Ω–∞–∏–±–æ–ª—å—à–∏–º –≤–µ—Å–æ–º

}
